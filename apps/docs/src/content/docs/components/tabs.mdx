## Usage

```tsx
import {
  Tabs,
  TabsContent,
  TabsIndicator,
  TabsList,
  TabsTrigger,
} from "@components/ui/v4/tabs"
```

```tsx
<Tabs>
  <TabsList>
    <TabsTrigger />
    <TabsTrigger />
    <TabsIndicator />
  </TabsList>
  <TabsContent />
  <TabsContent />
</Tabs>
```
