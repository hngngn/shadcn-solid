## Usage

```tsx
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuPortal,
  DropdownMenuTrigger,
} from "@components/ui/dropdown-menu"
```

```tsx
<DropdownMenu>
  <DropdownMenuTrigger />
  <DropdownMenuPortal>
    <DropdownMenuContent>
      <DropdownMenuItem />
    </DropdownMenuContent>
  </DropdownMenuPortal>
</DropdownMenu>
```
