## Usage

```tsx

  Slider,
  SliderFill,
  SliderThumb,
  SliderTrack,
} from "@components/ui/slider"
```

```tsx

  <SliderTrack>
    <SliderFill />
    <SliderThumb />
  </SliderTrack>
</Slider>
```
