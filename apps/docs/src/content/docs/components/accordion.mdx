## Usage

```tsx
import {
  Accordion,
  AccordionContent,
  AccordionItem,
  AccordionTrigger,
} from "@/components/ui/accordion"
```

```tsx
<Accordion>
  <AccordionItem value="item-1">
    <AccordionTrigger />
    <AccordionContent />
  </AccordionItem>
  <AccordionItem value="item-2">
    <AccordionTrigger />
    <AccordionContent />
  </AccordionItem>
</Accordion>
```
