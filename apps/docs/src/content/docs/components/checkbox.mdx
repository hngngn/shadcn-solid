## Usage

```tsx

  Checkbox,
  CheckboxControl,
  CheckboxDescription,
  CheckboxLabel,
} from "@/components/ui/checkbox"
```

```tsx

    <CheckboxDescription />
  </CheckboxControl>
</Checkbox>
```
